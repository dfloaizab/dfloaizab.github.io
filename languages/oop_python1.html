Abstracción
Mostrar solo lo esencial, ocultar la complejidad interna

¿Qué es?

Abstracción es el arte de definir qué hace algo sin revelar cómo lo hace. En Python se logra con clases abstractas: contrato que obliga a las subclases a implementar ciertos métodos, sin importar su lógica interna. Es como un contrato laboral: dice "debes saber cocinar" sin especificar la receta.

📐

Contrato claro

Defines la interfaz una vez. Cualquier subclase debe cumplirla.

🔄

Intercambiable

Puedes cambiar la implementación sin romper el código que la usa.

🧠

Reduce carga mental

El usuario solo necesita saber qué métodos existen, no su interior.

🛡️

Previene errores

Python lanza error si olvidas implementar un método abstracto.

Ejemplo 1 — Formas geométricas
abstraccion_formas.py Python 3 
from abc import ABC, abstractmethod
import math

# Clase abstracta: define QUÉ deben poder hacer las formas
class Forma(ABC):

    @abstractmethod
    def area(self) -> float:
        """Retorna el área de la forma."""
        ...

    @abstractmethod
    def perimetro(self) -> float:
        """Retorna el perímetro de la forma."""
        ...

    # Método concreto: todos comparten esta lógica
    def describir(self) -> str:
        return (f"{type(self).__name__}: "
                f"área={self.area():.2f}, perímetro={self.perimetro():.2f}")


class Circulo(Forma):
    def __init__(self, radio: float):
        self.radio = radio

    def area(self) -> float:
        return math.pi * self.radio ** 2

    def perimetro(self) -> float:
        return 2 * math.pi * self.radio


class Rectangulo(Forma):
    def __init__(self, ancho: float, alto: float):
        self.ancho = ancho
        self.alto = alto

    def area(self) -> float:
        return self.ancho * self.alto

    def perimetro(self) -> float:
        return 2 * (self.ancho + self.alto)


# Uso: el código no necesita saber qué forma concreta es
formas: list[Forma] = [Circulo(5), Rectangulo(4, 6), Circulo(2.5)]

for f in formas:
    print(f.describir())
# Circulo: área=78.54, perímetro=31.42
# Rectangulo: área=24.00, perímetro=20.00
# Circulo: área=19.63, perímetro=15.71
Ejemplo 2 — Notificaciones (email, SMS, push)
abstraccion_notificaciones.py Python 3 
from abc import ABC, abstractmethod
from dataclasses import dataclass

@dataclass
class Mensaje:
    destinatario: str
    asunto: str
    cuerpo: str


# Contrato: todo canal de notificación debe poder enviar
class CanalNotificacion(ABC):

    @abstractmethod
    def enviar(self, mensaje: Mensaje) -> bool:
        """Envía el mensaje. Retorna True si tuvo éxito."""
        ...

    @abstractmethod
    def nombre_canal(self) -> str:
        ...

    def enviar_con_log(self, mensaje: Mensaje) -> None:
        exito = self.enviar(mensaje)
        estado = "✓ enviado" if exito else "✗ falló"
        print(f"[{self.nombre_canal()}] {estado} → {mensaje.destinatario}")


class NotificacionEmail(CanalNotificacion):
    def enviar(self, mensaje: Mensaje) -> bool:
        # Aquí iría la integración real con SMTP
        print(f"  Enviando email a {mensaje.destinatario}: {mensaje.asunto}")
        return True

    def nombre_canal(self) -> str:
        return "EMAIL"


class NotificacionSMS(CanalNotificacion):
    def enviar(self, mensaje: Mensaje) -> bool:
        print(f"  SMS a {mensaje.destinatario}: {mensaje.cuerpo[:160]}")
        return True

    def nombre_canal(self) -> str:
        return "SMS"


# El sistema de alertas no importa qué canal usa
class SistemaAlertas:
    def __init__(self, canales: list[CanalNotificacion]):
        self.canales = canales

    def alertar(self, mensaje: Mensaje) -> None:
        for canal in self.canales:
            canal.enviar_con_log(mensaje)


sistema = SistemaAlertas([NotificacionEmail(), NotificacionSMS()])
alerta = Mensaje("usuario@mail.com", "Servidor caído", "El servidor web no responde.")
sistema.alertar(alerta)

Encapsulamiento
Controlar qué se expone y qué se protege del exterior

¿Qué es?

Encapsulamiento significa que los datos internos de un objeto no deben ser modificados libremente desde fuera. En Python no existe "privado" real como en Java, pero la convención _atributo indica "interno" y __atributo activa el name mangling. Los @property permiten lógica de validación al leer o escribir.

Convenciones en Python:
nombre → público, úsalo libremente
_nombre → "privado por convención", evita usarlo desde fuera
__nombre → name mangling: se renombra a _Clase__nombre, dificulta acceso externo
@property → acceso controlado con lógica de validación
Ejemplo 1 — Cuenta bancaria con validaciones
encapsulamiento_banco.py Python 3 
from datetime import datetime

class CuentaBancaria:
    def __init__(self, titular: str, saldo_inicial: float = 0.0):
        self._titular = titular
        self.__saldo = 0.0          # doble guion = name mangling
        self.__historial: list = []
        if saldo_inicial > 0:
            self.__depositar_interno(saldo_inicial, "Depósito inicial")

    # ── Lectura controlada ──────────────────
    @property
    def saldo(self) -> float:
        return self.__saldo

    @property
    def titular(self) -> str:
        return self._titular

    # ── Operaciones públicas ────────────────
    def depositar(self, monto: float) -> None:
        if monto <= 0:
            raise ValueError("El monto debe ser positivo")
        self.__depositar_interno(monto, "Depósito")

    def retirar(self, monto: float) -> None:
        if monto <= 0:
            raise ValueError("El monto debe ser positivo")
        if monto > self.__saldo:
            raise ValueError(f"Saldo insuficiente (tienes {self.__saldo:.2f})")
        self.__saldo -= monto
        self.__registrar(f"Retiro: -${monto:.2f}")

    def extrato(self) -> None:
        print(f"\n── Cuenta de {self._titular} ──")
        for entrada in self.__historial:
            print(f"  {entrada}")
        print(f"  Saldo actual: ${self.__saldo:.2f}\n")

    # ── Métodos internos (no usar desde fuera) ──
    def __depositar_interno(self, monto: float, concepto: str) -> None:
        self.__saldo += monto
        self.__registrar(f"{concepto}: +${monto:.2f}")

    def __registrar(self, texto: str) -> None:
        ts = datetime.now().strftime("%H:%M:%S")
        self.__historial.append(f"[{ts}] {texto}")


cuenta = CuentaBancaria("Ana García", 1000.0)
cuenta.depositar(500)
cuenta.retirar(200)
cuenta.extrato()

# cuenta.__saldo = 999999  ← esto NO funciona (name mangling)
# cuenta.saldo = 999999    ← AttributeError: no hay setter
print(cuenta.saldo)  # lectura OK → 1300.0
Ejemplo 2 — Temperatura con validación
encapsulamiento_temperatura.py Python 3 
class Termostato:
    """Controla temperatura entre límites físicos."""
    TEMP_MIN = -273.15  # cero absoluto
    TEMP_MAX = 1000.0

    def __init__(self, temp_inicial: float = 20.0):
        self.__temperatura = None
        self.temperatura = temp_inicial  # usa el setter

    @property
    def temperatura(self) -> float:
        return self.__temperatura

    @temperatura.setter
    def temperatura(self, valor: float) -> None:
        if not (self.TEMP_MIN <= valor <= self.TEMP_MAX):
            raise ValueError(
                f"Temperatura fuera de rango: {valor}°C"
                f" (válido: {self.TEMP_MIN}–{self.TEMP_MAX})"
            )
        self.__temperatura = valor

    @property
    def fahrenheit(self) -> float:
        return self.__temperatura * 9/5 + 32

    @property
    def estado(self) -> str:
        t = self.__temperatura
        if t < 0:      return "🧊 congelado"
        elif t < 20:   return "❄️  frío"
        elif t < 35:   return "✅ confortable"
        elif t < 60:   return "🔥 caliente"
        else:          return "💀 peligroso"

    def __repr__(self) -> str:
        return (f"Termostato({self.__temperatura:.1f}°C / "
                f"{self.fahrenheit:.1f}°F) → {self.estado}")


t = Termostato(22.0)
print(t)                # Termostato(22.0°C / 71.6°F) → ✅ confortable
t.temperatura = 37.5
print(t.estado)         # 🔥 caliente

try:
    t.temperatura = -300  # lanza ValueError
except ValueError as e:
    print(f"Error: {e}")

Herencia
Reutilizar y especializar comportamiento sin repetir código

¿Qué es?

Herencia permite que una clase adquiera atributos y métodos de otra. La subclase puede usar todo lo heredado, sobreescribir lo que necesite con super(), y agregar comportamiento nuevo. Python soporta herencia múltiple, pero úsala con cuidado.

♻️

Reutilización

Escribe la lógica una vez en la clase padre. Las hijas la heredan gratis.

🎯

Especialización

Sobreescribe solo lo que cambia. El resto se hereda sin cambios.

🏗️

Jerarquía clara

El árbol de clases documenta las relaciones "es un tipo de".

🔧

super()

Extiende el constructor padre sin duplicar código de inicialización.

Ejemplo 1 — Empleados con jerarquía
herencia_empleados.py Python 3 
from dataclasses import dataclass, field
from datetime import date

class Empleado:
    """Base: todo empleado tiene nombre, salario e ID."""

    _contador_ids: int = 0

    def __init__(self, nombre: str, salario_base: float):
        Empleado._contador_ids += 1
        self.id = Empleado._contador_ids
        self.nombre = nombre
        self._salario_base = salario_base
        self.fecha_ingreso = date.today()

    def calcular_salario(self) -> float:
        return self._salario_base

    def info(self) -> str:
        return (f"[{self.id}] {self.nombre} | "
                f"Salario: ${self.calcular_salario():,.0f}")

    def __repr__(self) -> str:
        return f"{type(self).__name__}({self.nombre!r})"


class EmpleadoComision(Empleado):
    """Empleado de ventas: salario base + comisión."""

    def __init__(self, nombre: str, salario_base: float, ventas: float, tasa_comision: float = 0.10):
        super().__init__(nombre, salario_base)  # llama al padre
        self.ventas = ventas
        self.tasa_comision = tasa_comision

    def calcular_salario(self) -> float:
        # Sobreescribe el método del padre y lo extiende
        return super().calcular_salario() + (self.ventas * self.tasa_comision)


class Gerente(Empleado):
    """Gerente: salario base + bono + equipo a cargo."""

    def __init__(self, nombre: str, salario_base: float, bono: float):
        super().__init__(nombre, salario_base)
        self.bono = bono
        self.equipo: list[Empleado] = []

    def contratar(self, empleado: Empleado) -> None:
        self.equipo.append(empleado)
        print(f"  {empleado.nombre} fue contratado por {self.nombre}")

    def calcular_salario(self) -> float:
        return super().calcular_salario() + self.bono

    def info(self) -> str:
        base = super().info()  # reutiliza el info del padre
        return f"{base} | Equipo: {len(self.equipo)} personas"


# Uso
gerente = Gerente("Laura Torres", 5000, bono=2000)
vendedor = EmpleadoComision("Carlos Ríos", 1500, ventas=20000)
analista = Empleado("María Pérez", 2500)

gerente.contratar(vendedor)
gerente.contratar(analista)

for emp in [gerente, vendedor, analista]:
    print(emp.info())
Ejemplo 2 — Vehículos con herencia múltiple
herencia_vehiculos.py Python 3 
class Vehiculo:
    def __init__(self, marca: str, modelo: str, anio: int):
        self.marca = marca
        self.modelo = modelo
        self.anio = anio
        self._encendido = False

    def encender(self) -> str:
        self._encendido = True
        return f"{self.marca} {self.modelo}: encendido 🟢"

    def apagar(self) -> str:
        self._encendido = False
        return f"{self.marca} {self.modelo}: apagado 🔴"

    def __str__(self) -> str:
        return f"{self.anio} {self.marca} {self.modelo}"


class Electrico:
    """Mixin: comportamiento de vehículo eléctrico."""
    def __init__(self, capacidad_bateria: float, **kwargs):
        super().__init__(**kwargs)  # MRO: pasa args al siguiente
        self.capacidad_bateria = capacidad_bateria
        self._carga = 100.0

    def cargar(self) -> str:
        self._carga = 100.0
        return f"Batería cargada al 100% ({self.capacidad_bateria} kWh)"

    def autonomia(self) -> float:
        return self.capacidad_bateria * self._carga / 100 * 6


class AutoElectrico(Electrico, Vehiculo):
    """Herencia múltiple: es un Vehículo y es Eléctrico."""
    def __init__(self, marca, modelo, anio, capacidad_bateria, puertas=4):
        super().__init__(
            marca=marca, modelo=modelo, anio=anio,
            capacidad_bateria=capacidad_bateria
        )
        self.puertas = puertas

    def encender(self) -> str:
        base = super().encender()
        return f"{base} | Silencioso ⚡"


tesla = AutoElectrico("Tesla", "Model 3", 2024, capacidad_bateria=75)
print(tesla.encender())
print(f"Autonomía: {tesla.autonomia():.0f} km")

Polimorfismo
El mismo mensaje, comportamientos distintos según quien lo recibe

¿Qué es?

Polimorfismo significa que objetos de distintas clases responden al mismo mensaje de forma diferente. Python lo implementa de dos maneras: polimorfismo por herencia (sobreescribir métodos) y duck typing ("si camina como pato y grazna como pato, es un pato") — no hace falta heredar de la misma clase si implementan la misma interfaz.

Tipo Mecanismo Cuándo usarlo
Por herencia Sobreescribir métodos de clase padre Jerarquías formales, clases abstractas
Duck typing Cualquier objeto con el método correcto Flexibilidad máxima, protocolos implícitos
Dunder methods __len__, __add__, __str__... Integración con sintaxis Python nativa
Ejemplo 1 — Renderer de documentos
polimorfismo_renderer.py Python 3 
from abc import ABC, abstractmethod

class Elemento(ABC):
    @abstractmethod
    def renderizar(self) -> str:
        ...

# Cada elemento sabe cómo renderizarse
class Titulo(Elemento):
    def __init__(self, texto: str, nivel: int = 1):
        self.texto = texto
        self.nivel = nivel
    def renderizar(self) -> str:
        prefijo = "#" * self.nivel
        return f"{prefijo} {self.texto}"

class Parrafo(Elemento):
    def __init__(self, texto: str):
        self.texto = texto
    def renderizar(self) -> str:
        return self.texto

class ListaElementos(Elemento):
    def __init__(self, items: list[str], ordenada: bool = False):
        self.items = items
        self.ordenada = ordenada
    def renderizar(self) -> str:
        if self.ordenada:
            lineas = [f"  {i+1}. {item}" for i, item in enumerate(self.items)]
        else:
            lineas = [f"  • {item}" for item in self.items]
        return "\n".join(lineas)

class Separador(Elemento):
    def renderizar(self) -> str:
        return "─" * 40

# El documento no sabe qué elementos tiene — solo los renderiza
class Documento:
    def __init__(self):
        self.elementos: list[Elemento] = []

    def agregar(self, *elems: Elemento) -> "Documento":
        self.elementos.extend(elems)
        return self  # permite encadenamiento

    def renderizar(self) -> str:
        # Polimorfismo: llamamos el mismo método en objetos distintos
        return "\n\n".join(e.renderizar() for e in self.elementos)


doc = (Documento()
    .agregar(Titulo("Mi Reporte"))
    .agregar(Parrafo("Este es el resumen ejecutivo."))
    .agregar(Separador())
    .agregar(Titulo("Hallazgos", nivel=2))
    .agregar(ListaElementos(["Aumento de ventas 30%", "Costos reducidos 15%"]))
)
print(doc.renderizar())
Ejemplo 2 — Duck typing y operadores sobrecargados
polimorfismo_duck_typing.py Python 3 
# Duck typing: no necesitan heredar de la misma clase
class Vector2D:
    def __init__(self, x: float, y: float):
        self.x = x
        self.y = y

    def __add__(self, otro: "Vector2D") -> "Vector2D":
        return Vector2D(self.x + otro.x, self.y + otro.y)

    def __mul__(self, escalar: float) -> "Vector2D":
        return Vector2D(self.x * escalar, self.y * escalar)

    def __eq__(self, otro: object) -> bool:
        if not isinstance(otro, Vector2D): return NotImplemented
        return self.x == otro.x and self.y == otro.y

    def __abs__(self) -> float:
        return (self.x**2 + self.y**2) ** 0.5

    def __repr__(self) -> str:
        return f"Vector2D({self.x}, {self.y})"

    def __len__(self) -> int:
        return 2  # dimensiones


v1 = Vector2D(3, 4)
v2 = Vector2D(1, 2)

print(v1 + v2)       # Vector2D(4, 6)   — usa __add__
print(v1 * 2)        # Vector2D(6, 8)   — usa __mul__
print(abs(v1))       # 5.0              — usa __abs__
print(len(v1))       # 2                — usa __len__
print(v1 == v2)      # False            — usa __eq__

# Duck typing: esta función acepta CUALQUIER cosa con .renderizar()
def imprimir_todo(items) -> None:
    for item in items:
        print(item.renderizar())  # no importa el tipo

# Incluso puede mezclarse con clases sin relación de herencia
class MensajeSimple:
    def __init__(self, texto): self.texto = texto
    def renderizar(self) -> str: return f">> {self.texto}"

imprimir_todo([Titulo("Hola"), MensajeSimple("Sin herencia, funciona igual")])

S — Responsabilidad Única
Single Responsibility Principle
Single Responsibility Principle

Regla

Una clase debe tener una sola razón para cambiar. Si una clase hace demasiadas cosas, un cambio en una funcionalidad puede romper otra. El truco mental: si describes la clase con "y" en la mitad ("guarda datos Y envía emails"), la estás dividiendo.

✗ Violación SRP

    # Hace TRES cosas: datos, DB y email
    def __init__(self, items):
        self.items = items

    def calcular_total(self): ...

    def guardar_en_db(self): ...

    def enviar_confirmacion(self): ...
✓ Aplicando SRP
class Pedido:      # solo datos
    def calcular_total(self): ...

class RepositorioPedido:  # solo DB
    def guardar(self, pedido): ...

class EmailPedido:  # solo email
    def confirmar(self, pedido): ...

from dataclasses import dataclass, field
from typing import Protocol

@dataclass
class LineaFactura:
    descripcion: str
    cantidad: int
    precio_unitario: float

    @property
    def subtotal(self) -> float:
        return self.cantidad * self.precio_unitario


# Clase 1: solo representa la factura y su lógica de negocio
@dataclass
class Factura:
    numero: str
    cliente: str
    lineas: list[LineaFactura] = field(default_factory=list)

    def total(self) -> float:
        return sum(l.subtotal for l in self.lineas)

    def agregar_linea(self, linea: LineaFactura) -> None:
        self.lineas.append(linea)


# Clase 2: solo se encarga de imprimir/formatear
class ImpresoraFactura:
    def imprimir(self, factura: Factura) -> str:
        lineas = [f"FACTURA #{factura.numero}",
                  f"Cliente: {factura.cliente}",
                  "-" * 30]
        for l in factura.lineas:
            lineas.append(f"  {l.descripcion:{20}} ${l.subtotal:>8.2f}")
        lineas.append(f"  {'TOTAL':{20}} ${factura.total():>8.2f}")
        return "\n".join(lineas)


# Clase 3: solo se encarga de persistencia
class RepositorioFactura:
    def __init__(self):
        self._almacen: dict[str, Factura] = {}

    def guardar(self, factura: Factura) -> None:
        self._almacen[factura.numero] = factura

    def buscar(self, numero: str) -> Factura | None:
        return self._almacen.get(numero)


# Uso
factura = Factura("F-001", "Tech Corp")
factura.agregar_linea(LineaFactura("Laptop Pro", 2, 1500.00))
factura.agregar_linea(LineaFactura("Mouse inalámbrico", 3, 45.00))

repo = RepositorioFactura()
repo.guardar(factura)

impresora = ImpresoraFactura()
print(impresora.imprimir(factura))

Open/Closed Principle
Open/Closed Principle

Regla

Abierto para extensión, cerrado para modificación. Agregar nueva funcionalidad no debe requerir editar código existente. Se logra con abstracciones y polimorfismo: defines el contrato una vez, añades nuevas implementaciones sin tocar las existentes.

from abc import ABC, abstractmethod
import json

class Exportador(ABC):
    @abstractmethod
    def exportar(self, datos: list[dict]) -> str: ...

class ExportadorCSV(Exportador):
    def exportar(self, datos: list[dict]) -> str:
        if not datos: return ""
        cabecera = ",".join(datos[0].keys())
        filas = [",".join(str(v) for v in d.values()) for d in datos]
        return "\n".join([cabecera] + filas)

class ExportadorJSON(Exportador):
    def exportar(self, datos: list[dict]) -> str:
        return json.dumps(datos, ensure_ascii=False, indent=2)

class ExportadorMarkdown(Exportador):
    def exportar(self, datos: list[dict]) -> str:
        if not datos: return ""
        cols = list(datos[0].keys())
        sep = " | "
        cabecera = sep.join(cols)
        divisor = sep.join(["---"] * len(cols))
        filas = [sep.join(str(d[c]) for c in cols) for d in datos]
        return "\n".join([cabecera, divisor] + filas)

# Servicio de reportes: nunca cambia aunque agreguemos exportadores
class ServicioReportes:
    def __init__(self, exportador: Exportador):
        self.exportador = exportador

    def generar(self, datos: list[dict]) -> str:
        return self.exportador.exportar(datos)


ventas = [
    {"producto": "Laptop", "cantidad": 3, "total": 4500},
    {"producto": "Mouse",  "cantidad": 8, "total": 360},
]

for exp in [ExportadorCSV(), ExportadorJSON(), ExportadorMarkdown()]:
    reporte = ServicioReportes(exp)
    print(f"\n── {type(exp).__name__} ──")
    print(reporte.generar(ventas))

L — Principio de Liskov
Liskov Substitution Principle
Liskov Substitution Principle

Regla

Una subclase debe poder reemplazar a su clase padre sin romper el programa. Si en algún punto tienes que preguntar "¿es un Perro o un Gato?" para comportarte diferente, algo está mal. El ejemplo clásico: un Cuadrado hereda de Rectangulo — parece lógico, pero viola LSP.

Ejemplo — El problema del Cuadrado/Rectángulo y su solución
lsp_formas.pyPython 3
# ✗ VIOLACIÓN: Cuadrado hereda de Rectángulo
class RectanguloMalo:
    def __init__(self, ancho, alto):
        self.ancho = ancho
        self.alto = alto

class CuadradoMalo(RectanguloMalo):
    def __init__(self, lado):
        super().__init__(lado, lado)

    # Problema: cambiar ancho obliga a cambiar alto
    @RectanguloMalo.ancho.setter
    def ancho(self, v): self._ancho = self._alto = v


def duplicar_ancho(r: RectanguloMalo):
    r.ancho *= 2
    print(f"Esperado alto=5, obtenido: {r.alto}")  # ¡CuadradoMalo devuelve 10!


# ✓ SOLUCIÓN: jerarquía correcta con abstracción compartida
from abc import ABC, abstractmethod

class Forma(ABC):
    @abstractmethod
    def area(self) -> float: ...
    @abstractmethod
    def perimetro(self) -> float: ...

class Rectangulo(Forma):
    def __init__(self, ancho: float, alto: float):
        self.ancho = ancho
        self.alto = alto
    def area(self) -> float: return self.ancho * self.alto
    def perimetro(self) -> float: return 2 * (self.ancho + self.alto)

class Cuadrado(Forma):  # hermano de Rectangulo, no hijo
    def __init__(self, lado: float):
        self.lado = lado
    def area(self) -> float: return self.lado ** 2
    def perimetro(self) -> float: return 4 * self.lado

# Ahora sí — cualquier Forma puede sustituir a otra en esta función
def describir_forma(f: Forma) -> None:
    print(f"{type(f).__name__}: área={f.area():.2f}, perímetro={f.perimetro():.2f}")

for forma in [Rectangulo(4, 5), Cuadrado(4), Rectangulo(10, 2)]:
    describir_forma(forma)  # funciona perfectamente para todos

I — Segregación de Interfaces
Interface Segregation Principle
Interface Segregation Principle

Regla

Ninguna clase debe depender de métodos que no usa. Prefiere muchas interfaces pequeñas sobre una interfaz grande. Si una clase implementa una interfaz y tiene que dejar métodos vacíos o lanzar NotImplementedError, es señal de que la interfaz es demasiado grande.

✗ Interfaz demasiado grande
class Trabajador(ABC):
    @abstractmethod
    def trabajar(self): ...
    @abstractmethod
    def comer(self): ...
    @abstractmethod
    def dormir(self): ...

class Robot(Trabajador):
    def trabajar(self): ...
    def comer(self):   pass  # ¡forzado!
    def dormir(self):  pass  # ¡forzado!
✓ Interfaces pequeñas
class Laborable(ABC):
    @abstractmethod
    def trabajar(self): ...

class Biologico(ABC):
    @abstractmethod
    def comer(self): ...
    @abstractmethod
    def dormir(self): ...

class Robot(Laborable): ...   # solo trabaja
class Humano(Laborable, Biologico): ...

from abc import ABC, abstractmethod

# Interfaces pequeñas y enfocadas
class Imprimible(ABC):
    @abstractmethod
    def imprimir(self, documento: str) -> None: ...

class Escaneable(ABC):
    @abstractmethod
    def escanear(self) -> str: ...

class Faxeable(ABC):
    @abstractmethod
    def enviar_fax(self, numero: str, doc: str) -> None: ...


# Cada dispositivo implementa solo lo que puede hacer
class ImpresoraBasica(Imprimible):
    def imprimir(self, documento: str) -> None:
        print(f"[ImpresoraBasica] Imprimiendo: {documento[:40]}...")


class EscanerSimple(Escaneable):
    def escanear(self) -> str:
        return "[EscanerSimple] Documento escaneado"


class ImpresoraMultifuncion(Imprimible, Escaneable, Faxeable):
    def imprimir(self, documento: str) -> None:
        print(f"[Multifunción] Imprimiendo en alta calidad: {documento[:30]}")

    def escanear(self) -> str:
        return "[Multifunción] Escaneado a 600 DPI"

    def enviar_fax(self, numero: str, doc: str) -> None:
        print(f"[Multifunción] Fax enviado a {numero}")


# Funciones que dependen solo de la interfaz que necesitan
def imprimir_reporte(imp: Imprimible, reporte: str) -> None:
    imp.imprimir(reporte)

def digitalizar(scan: Escaneable) -> str:
    return scan.escanear()


impresora = ImpresoraBasica()
multifuncion = ImpresoraMultifuncion()

imprimir_reporte(impresora, "Reporte Q3 2024")
imprimir_reporte(multifuncion, "Reporte Q3 2024")
print(digitalizar(multifuncion))

D — Inversión de Dependencias
Dependency Inversion Principle
Dependency Inversion Principle

Regla

Los módulos de alto nivel no deben depender de módulos de bajo nivel. Ambos deben depender de abstracciones. Esto se implementa con inyección de dependencias: en lugar de que una clase cree sus propias dependencias con self.db = MySQL(), las recibe desde fuera. Facilita enormemente el testing.

✓ Dependencia inyectada
class ServicioUsuario:
    def __init__(self, db: BaseDatos,
                       email: ServicioEmail):
        self.db = db
        self.email = email
    # Fácil de testear con mocks

from abc import ABC, abstractmethod

# Abstracciones (contratos)
class RepositorioUsuarios(ABC):
    @abstractmethod
    def buscar_por_id(self, uid: int) -> dict | None: ...
    @abstractmethod
    def guardar(self, usuario: dict) -> None: ...

class ServicioEmail(ABC):
    @abstractmethod
    def enviar(self, destino: str, asunto: str, cuerpo: str) -> bool: ...


# Implementaciones de bajo nivel
class RepositorioMemoria(RepositorioUsuarios):
    def __init__(self):
        self._datos: dict[int, dict] = {}

    def buscar_por_id(self, uid: int) -> dict | None:
        return self._datos.get(uid)

    def guardar(self, usuario: dict) -> None:
        self._datos[usuario["id"]] = usuario


class EmailConsola(ServicioEmail):
    """Para desarrollo/testing: imprime en consola."""
    def enviar(self, destino: str, asunto: str, cuerpo: str) -> bool:
        print(f"\n📧 Para: {destino}\n   Asunto: {asunto}\n   {cuerpo}")
        return True


# Módulo de alto nivel: depende SOLO de abstracciones
class ServicioRegistro:
    def __init__(self,
                 repo: RepositorioUsuarios,   # abstracción
                 email: ServicioEmail):          # abstracción
        self._repo = repo
        self._email = email
        self._siguiente_id = 1

    def registrar(self, nombre: str, email_addr: str) -> dict:
        usuario = {"id": self._siguiente_id, "nombre": nombre, "email": email_addr}
        self._siguiente_id += 1
        self._repo.guardar(usuario)
        self._email.enviar(
            email_addr,
            "Bienvenido",
            f"Hola {nombre}, tu cuenta ha sido creada."
        )
        return usuario


# Producción
repo = RepositorioMemoria()
mailer = EmailConsola()
servicio = ServicioRegistro(repo, mailer)  # inyección

u = servicio.registrar("Ana García", "ana@ejemplo.com")
print(repo.buscar_por_id(u["id"]))

# Testing: reemplaza dependencias con mocks sin tocar ServicioRegistro
class EmailMock(ServicioEmail):
    def __init__(self): self.enviados = []
    def enviar(self, d, a, c) -> bool:
        self.enviados.append({"destino": d, "asunto": a})
        return True

mock_email = EmailMock()
servicio_test = ServicioRegistro(RepositorioMemoria(), mock_email)
servicio_test.registrar("Test User", "test@mail.com")
print(mock_email.enviados)  # [{'destino': 'test@mail.com', 'asunto': 'Bienvenido'}]
