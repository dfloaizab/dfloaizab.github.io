Programación Orientada a Objetos

Abstracción

Permite modelar conceptos del dominio mediante interfaces independientes de implementación.

from abc import ABC, abstractmethod class MetodoPago(ABC): @abstractmethod def pagar(self, monto): pass class TarjetaCredito(MetodoPago): def pagar(self, monto): return f"Pago {monto} con tarjeta"

Encapsulamiento

class Cuenta: def __init__(self): self.__saldo = 0 def depositar(self, m): if m > 0: self.__saldo += m

Herencia

class Empleado: def salario(self): pass class Ingeniero(Empleado): def salario(self): return 5000

Polimorfismo

def nomina(empleados): return sum(e.salario() for e in empleados)
